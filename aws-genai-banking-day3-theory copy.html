boto3 Fundamentals for Banking

üè¶ SecureBank Python Automation Strategy

Client vs Resource APIs:

  • Client (Low-level): Direct API mapping, more control
  • Resource (High-level): Object-oriented, easier to use
  • Banking Use: Client for complex operations, Resource for simple tasks
  • Performance: Client is faster for bulk operations

Authentication Methods:

  • IAM Roles: Recommended for EC2 instances
  • Environment Variables: For local development
  • Credentials File: For multiple profiles
  • STS Tokens: For temporary access

S3 Operations for Banking Documents

import boto3 import logging from botocore.exceptions import ClientError # Configure logging logging.basicConfig(level=logging.INFO) logger = logging.getLogger(__name__) class SecureBankS3Manager: def __init__(self, region_name='us-east-1'): self.s3_client = boto3.client('s3', region_name=region_name) self.s3_resource = boto3.resource('s3', region_name=region_name) def list_banking_buckets(self): """List all SecureBank S3 buckets""" try: response = self.s3_client.list_buckets() banking_buckets = [ bucket['Name'] for bucket in response['Buckets'] if 'securebank' in bucket['Name'].lower() ] logger.info(f"Found {len(banking_buckets)} banking buckets") return banking_buckets except ClientError as e: logger.error(f"Error listing buckets: {e}") return [] def upload_loan_document(self, file_path, bucket_name, key_name): """Upload loan document with encryption""" try: self.s3_client.upload_file( file_path, bucket_name, key_name, ExtraArgs={ 'ServerSideEncryption': 'AES256', 'Metadata': { 'document-type': 'loan-application', 'uploaded-by': 'banking-system', 'compliance': 'pci-dss' } } ) logger.info(f"Successfully uploaded {key_name} to {bucket_name}") return True except ClientError as e: logger.error(f"Error uploading file: {e}") return False def download_processed_documents(self, bucket_name, prefix='processed/'): """Download processed banking documents""" try: bucket = self.s3_resource.Bucket(bucket_name) for obj in bucket.objects.filter(Prefix=prefix): local_path = f"./downloads/{obj.key}" os.makedirs(os.path.dirname(local_path), exist_ok=True) bucket.download_file(obj.key, local_path) logger.info(f"Downloaded {obj.key}") except ClientError as e: logger.error(f"Error downloading files: {e}") # Usage example s3_manager = SecureBankS3Manager() buckets = s3_manager.list_banking_buckets() print(f"Banking buckets: {buckets}")

Error Handling and Best Practices

üõ°Ô∏è Robust Error Handling

import boto3 from botocore.exceptions import ( ClientError, NoCredentialsError, PartialCredentialsError, BotoCoreError ) import time import random def retry_with_backoff(func, max_retries=3): """Implement exponential backoff for AWS operations""" for attempt in range(max_retries): try: return func() except ClientError as e: error_code = e.response['Error']['Code'] if error_code in ['Throttling', 'RequestLimitExceeded']: wait_time = (2 ** attempt) + random.uniform(0, 1) logger.warning(f"Rate limited. Retrying in {wait_time:.2f}s") time.sleep(wait_time) elif error_code == 'NoSuchBucket': logger.error("Bucket does not exist") raise else: logger.error(f"Unexpected error: {e}") if attempt == max_retries - 1: raise except NoCredentialsError: logger.error("AWS credentials not found") raise except Exception as e: logger.error(f"Unexpected error: {e}") raise # Example usage def safe_s3_operation(): s3_client = boto3.client('s3') return retry_with_backoff( lambda: s3_client.list_objects_v2( Bucket='securebank-documents' ) )
